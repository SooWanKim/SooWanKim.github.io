### 문제1

```c++
#include <string>
#include <vector>
#include <math.h>

using namespace std;

struct NumberPoint
{
    int x;
    int y;
    NumberPoint(int left, int right)
    {
        x=left;
        y=right;
    }

    NumberPoint()
    {
        x=0;
        y=0;
    }

    int operator-(const NumberPoint& in)
    {
        return (abs(in.x - x) + abs(in.y - y));
    }
};

string solution(vector<int> numbers, string hand) {
    string answer = "";

    NumberPoint startLeft(3,0);
    NumberPoint startRight(3,2);

    for(int i=0 ; i<numbers.size() ; i++)
    {
        NumberPoint temp;
        bool isCloserLeft = true;
        int number = numbers[i];

        if(number < 0 || 9 < number)
            continue;

        if( number == 0)
        {
            temp.x=3;
            temp.y=1;
        }
        else
        {
            temp.x=(number-1) / 3;
            temp.y=number - ((temp.x) * 3 + 1);
        }
        if(temp.y == 0)
        {
            isCloserLeft = true;
        }
        else if(temp.y == 2)
        {
            isCloserLeft = false;
        }
        else
        {
            int leftDist = startLeft - temp;
            int rightDist = startRight - temp;

            if(leftDist == rightDist)
            {
                if(hand == "left")
                {
                    isCloserLeft = true;
                }
                else if(hand == "right")
                {
                    isCloserLeft = false;
                }
                else
                {
                    continue;
                }
            }
            else if(leftDist < rightDist)
            {
                isCloserLeft = true;
            }
            else
            {
                isCloserLeft = false;
            }
        }

        if(isCloserLeft == true)
        {
             startLeft = temp;
             answer+= "L";
        }
        else
        {
             startRight = temp;
             answer+= "R";
        }
    }

    return answer;
}
```
